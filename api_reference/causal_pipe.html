125    def _convert_causal_effect_method(self, method: CausalEffectMethod) -> CausalEffectMethod:
126        """Convert a generic ``CausalEffectMethod`` into a specific subclass."""
127
128        if method.__class__ is CausalEffectMethod:
129            params = dict(method.params or {})
130            common_args = {"directed": method.directed}
131            if method.name == CausalEffectMethodNameEnum.PEARSON:
132                return PearsonCausalEffectMethod(**common_args)
133            if method.name == CausalEffectMethodNameEnum.SPEARMAN:
134                return SpearmanCausalEffectMethod(**common_args)
135            if method.name == CausalEffectMethodNameEnum.MI:
136                return MICausalEffectMethod(**common_args)
137            if method.name == CausalEffectMethodNameEnum.KCI:
138                return KCICausalEffectMethod(**common_args)
139            if method.name == CausalEffectMethodNameEnum.SEM:
140                return SEMCausalEffectMethod(**common_args, **params)
141            if method.name == CausalEffectMethodNameEnum.SEM_CLIMBING:
142                return SEMClimbingCausalEffectMethod(**common_args, **params)
143            if method.name == CausalEffectMethodNameEnum.PYSR:
144                known = {
145                    k: params.pop(k)
146                    for k in [
147                        "noise_kind",
148                        "alpha",
149                        "tol",
150                        "max_iter",
151                        "restarts",
152                        "standardized_init",
153                        "hc_orient_undirected_edges",
154                    ]
155                    if k in params
156                }
157                return PYSRCausalEffectMethod(
158                    **common_args, pysr_params=params, **known
159                )
160        return method

507        except Exception as e:
